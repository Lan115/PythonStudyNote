## 元组 - 不可变序列

### 特点：创建后不能修改

```
# 创建
t1 = (1, 2, 3)
t2 = tuple([4, 5, 6])
t3 = 1, 2, 3  # 括号可省略

# 查 - 类似列表
print(t1[0])      # 1
print(t1[1:3])    # (2, 3)
print(2 in t1)    # True

# 增 - 不能直接添加，但可以连接
t4 = t1 + (4,)    # (1, 2, 3, 4)

# 删 - 不能删除元素，只能删除整个元组
del t1  # 删除变量

# 改 - 不能修改元素
# t1[0] = 10  # TypeError

# 其他操作
count = t1.count(2)  # 统计出现次数
index = t1.index(2)  # 查找索引
```

## 字典- 键值对集合

### 增

```
# 创建
d1 = {'name': 'Alice', 'age': 25}
d2 = dict(name='Bob', age=30)

# 添加/修改
d1['city'] = 'Beijing'      # 添加新键值对
d1['age'] = 26              # 修改已有键的值
d1.update({'gender': 'F', 'age': 27})  # 批量更新
```

### 删

```
del d1['city']           # 删除指定键
value = d1.pop('age')    # 删除并返回值
item = d1.popitem()      # 删除并返回最后一个键值对
d1.clear()               # 清空字典
# del d1                 # 删除整个字典
```

### 改

```
d1['name'] = 'Carol'     # 直接赋值修改
d1.update({'name': 'David', 'age': 28})  # 批量修改
```

### 查

```
print(d1['name'])        # 直接访问，键不存在会报错
print(d1.get('name'))    # 安全访问，返回None如果键不存在
print(d1.get('phone', 'N/A'))  # 提供默认值

# 遍历
for key in d1:           # 遍历键
    print(key, d1[key])

for key, value in d1.items():  # 遍历键值对
    print(key, value)

for value in d1.values():      # 遍历值
    print(value)

# 检查存在
print('name' in d1)      # True
```

## 集合- 无序不重复元素

### 增

```
# 创建
s1 = {1, 2, 3}
s2 = set([3, 4, 5])

# 添加元素
s1.add(4)           # {1, 2, 3, 4}
s1.update([5, 6])   # {1, 2, 3, 4, 5, 6} - 添加多个
```

### 删

```
s1.remove(3)        # 删除元素，不存在会报错 {1, 2, 4, 5, 6}
s1.discard(10)      # 安全删除，不存在不会报错
value = s1.pop()    # 随机删除并返回一个元素
s1.clear()          # 清空集合
# del s1            # 删除整个集合
```

### 改

```
# 集合元素不可修改，但可以进行集合运算
s1 = {1, 2, 3}
s2 = {3, 4, 5}

# 集合操作
union = s1 | s2        # 并集: {1, 2, 3, 4, 5}
intersection = s1 & s2 # 交集: {3}
difference = s1 - s2   # 差集: {1, 2}
symmetric_diff = s1 ^ s2  # 对称差集: {1, 2, 4, 5}
```

### 查

```
s1 = {1, 2, 3, 4, 5}

# 遍历
for item in s1:
    print(item)

# 检查存在
print(3 in s1)      # True
print(10 in s1)     # False

# 集合关系
s2 = {3, 4}
print(s2.issubset(s1))      # True - s2是s1的子集
print(s1.issuperset(s2))    # True - s1是s2的超集
print(s1.isdisjoint({6,7})) # True - 没有共同元素
```
